local function CreateWeld(part, Location, c0, type_)

	if part:IsA("Folder") then
		for _, v in ipairs(part:GetChildren()) do
			if v:IsA("Attachment") then
				v.Parent = Location
			end
		end
		part:Destroy()
		return nil
	end

	if not Location:FindFirstChild("Visuals") then
		local Visuals = Instance.new("Folder")
		Visuals.Name = "Visuals"
		Visuals.Parent = Location
	end

	local weld = Instance.new("Weld")
	weld.Part0 = Location
	weld.Part1 = part
	weld.C0 = c0
	weld.Parent = Location.Visuals

	part.Parent = Location
	part.Transparency = 1
	part.Anchored = false
	part.CanCollide= false
	part.CanTouch = false
	part.CanQuery = false
	part.Massless = true

	if type_ == "Cluster" then
		local tab = {}
		for _, v in ipairs(part:GetDescendants()) do
			if v:IsA("ParticleEmitter") then
				table.insert(tab, v)
			end
		end
		return tab
	end

	local Visual = part:FindFirstChild("Trail") or part:FindFirstChildOfClass("ParticleEmitter")
	Visual.Enabled = false
	return Visual

end

function AlignCharWithNormal(hitPosition, normal)
	if WallJump then return end
	local offset = 1
	local charPos = hitPosition + normal*offset
	local lookDirection = -normal
	Hrp.CFrame = CFrame.new(charPos, charPos + lookDirection)
end

local function CreateBodyVelocity(name, velocity, maxForce)
	local bodyVelocity = Instance.new("BodyVelocity")
	bodyVelocity.Name = name
	bodyVelocity.Velocity = velocity
	bodyVelocity.MaxForce = maxForce

local function GetVerticalSpeed(part)
	if part and part:IsA("BasePart") then
		return part.Velocity.Y
	end
	return 0
end

local function GetAllSpeed(part)
	if part and part:IsA("BasePart") then
		return part.Velocity.Magnitude
	end
end

local function GetHorizontalSpeed(part)
	if part and part:IsA("BasePart") then
		local velocity = part.Velocity
		local horizontalVector = Vector3.new(velocity.X, 0, velocity.Z)
		return horizontalVector.Magnitude
	end
	return 0
end

local function ChangeStamina()
	while true do
		task.wait(0.01)

		local curr = Player:GetAttribute("Stamina")
		if not curr then continue end

		if curr == 0 then
			States.rest = true
		elseif curr == 100 then
			States.rest = false
		end

		if States.running then
			Player:SetAttribute("Stamina", math.clamp(curr - 1, 0, 100))
		else
			Player:SetAttribute("Stamina", math.clamp(curr + 1, 0, 100))
		end
	end
end

local function EndRun()
	Hum.WalkSpeed = WalkSpeed
	States.running = false
	SetWalkSpeed(1)
end

local function StartRun()
	if Player:GetAttribute("Stamina") <= 0 then 
		EndRun()
		return 
	end
	Hum.WalkSpeed = RunSpeed
	States.running = true
	SetWalkSpeed(2)
end

local Jumpgap = 0
local canDjump = false

local function CreateJump(moveDir, impulse)
	local totalImpulse = moveDir + impulse
	AnimationTracks.JumpTrack:Play()
	Hrp:ApplyImpulse(totalImpulse)

	States.jumping = true
	CurrJumpPower -= FatiqueFactor
	lastJumped = tick()
end	

local function PlayJumpEffect(character)
	-- 1. Remove old highlight if they jump twice fast
	local oldHighlight = character:FindFirstChild("JumpPadEffect")
	if oldHighlight then oldHighlight:Destroy() end

	-- 2. Create the Highlight
	local highlight = Instance.new("Highlight")
	highlight.Name = "JumpPadEffect"
	highlight.Adornee = character
	highlight.Parent = character

	-- // CUSTOMIZE COLORS HERE //
	highlight.FillColor = Color3.fromRGB(100, 255, 255) -- Cyan (Electric look)
	highlight.OutlineColor = Color3.fromRGB(255, 255, 255) -- White outline

	-- Start bright
	highlight.FillTransparency = 0.2
	highlight.OutlineTransparency = 0

	-- 3. Tween it out (Fade away)
	local tweenInfo = TweenInfo.new(1.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
	local tween = TweenService:Create(highlight, tweenInfo, {
		FillTransparency = 1,
		OutlineTransparency = 1
	})

	tween:Play()

	-- 4. Clean up
	tween.Completed:Connect(function()
		highlight:Destroy()
	end)
end



local function HandleJump(input, phase)

	if input.KeyCode ~= Inputs.jumping then return end


	local FreeFallState = Enum.HumanoidStateType.Freefall

	--Double Jump
	if phase == "Began" 
		and Hum:GetState() == FreeFallState 
		and canDjump
		and not States.climbing 
	then
		--CreateJump(Hum.MoveDirection * 5, Vector3.new(0,CurrJumpPower,0) + Vector3.new(0, -math.clamp(verticleSpeed, -math.huge, 0) * 10 ,0))		
		Hum.JumpPower = 20
		Hum.JumpHeight = 20
		Hum:ChangeState(Enum.HumanoidStateType.Jumping)
		Player:SetAttribute("SoundType", "DJump")
		canDjump = false
		CreateJumpRing()
		Hum:ChangeState(Enum.HumanoidStateType.Freefall)
		Hum.JumpPower = 0
		Hum.JumpHeight = 0
		return
	end

	--Normal Jump
	if phase == "Began" 
		and Hum:GetState() ~= FreeFallState 
		and not States.jumping 
	then
		--CreateJump(Hum.MoveDirection * 5, Vector3.new(0,CurrJumpPower,0))
		canDjump = true
		Player:SetAttribute("SoundType", "Jump")
		return
	end

	--Wall Jump

	if phase == "Began" 
		and not States.jumping
		and States.climbing 
	then
		WallJump = true --To prevent aliging the player to wall again
		canDjump = false --can Double Jump
		CreateJump(-Hrp.CFrame.LookVector * 500, Vector3.new(0,CurrJumpPower,0))

		task.delay(1, function() 
			WallJump = false
			canDjump = true
		end)
		return
	end

end

RunService.RenderStepped:Connect(function()

	verticleSpeed = GetVerticalSpeed(Hrp)
	local HorizontalSpeed = GetHorizontalSpeed(Hrp)
	local HumState = Hum:GetState()

	local Rays = {
		Climb = {
			Head = Head.CFrame.LookVector * ClimbRayLen,
			Hrp = Hrp.CFrame.LookVector * ClimbRayLen,
			HrpDown = Hrp.CFrame.UpVector * -5
		}
	}

	local RaysCasted = 
		{
			ClimbHead = workspace:Raycast(Head.Position, Rays.Climb.Head, rayParams),
			ClimbHrp = workspace:Raycast(Hrp.Position, Rays.Climb.Hrp, rayParams),
			ClimbHrpDown = workspace:Raycast(Hrp.Position, Rays.Climb.HrpDown, rayParams)
		}

	--Climb only if certain distance above ground
	if not RaysCasted.ClimbHrpDown  then 

		if RaysCasted.ClimbHrp then
			AlignCharWithNormal(RaysCasted.ClimbHrp.Position, RaysCasted.ClimbHrp.Normal)
			States.jumping = false
			States.climbing = true
		else
			States.climbing = false
			AnimationTracks.WallClimb:Stop()
		end

	elseif RaysCasted.ClimbHrpDown.Instance.Name == "JumpPad" then

		if not States.PadJump  then
			States.PadJump = true
			States.PadJumpedAir = true
			local forwardDir = Hrp.CFrame.LookVector
			local upPower = 2000
			local forwardPower = 2000 
			local totalImpulse = Vector3.new(forwardDir.X * forwardPower, upPower, forwardDir.Z * forwardPower)
			Hum:ChangeState(Enum.HumanoidStateType.Jumping)
			Hrp:ApplyImpulse(totalImpulse)
			task.spawn(function()
				PlayJumpEffect(Character)
			end)
			Audios.JumpPad:Play()
			task.delay(1, function()
				States.PadJump = false
			end)
		end

	else 
		States.climbing = false
		AnimationTracks.WallClimb:Stop()

		if RaysCasted.ClimbHrpDown.Instance:GetAttribute("Lava") then
			local impulse = Vector3.new(0,800,0)
			Hum:TakeDamage(2)
			Hrp:ApplyImpulse(impulse)
			script.burn:Play()
			task.delay(0.1, function()
				ToggleLavaSmoke(true)
			end)
		else
			ToggleLavaSmoke(false)
			local gap = (RaysCasted.ClimbHrpDown.Position - Hrp.Position).Magnitude





			if verticleSpeed < -170 then --if falling fast then crash anim
				if States.PadJumpedAir then
					States.PadJumpedAir = false

				elseif math.abs(gap) < 5 then
					Injured = true
					StatsAndValues.ToggleInputs(true)
					AnimationTracks.FallCrash:Play()
					Player:SetAttribute("SoundType", "Crash")
					task.delay(1.5, function()
						Injured = false
						StatsAndValues.ToggleInputs(false)
					end)
				end	
			end
		end



	end


	Jumpgap = tick() - lastJumped

	--If running then smoke
	if HumState == Enum.HumanoidStateType.Running then
		if HorizontalSpeed > 30 then
			ToggleRunSmoke()
		end
	end

	if HumState == Enum.HumanoidStateType.Freefall then
		if not States.climbing then
			ToggleHandTrails(true)
		else
			ToggleHandTrails(false)
		end
	else
		ToggleHandTrails(false)
		AnimationTracks.DoubleJumpTrack:Stop()
	end

	if HumState ~= Enum.HumanoidStateType.Freefall and Jumpgap > 0.5 then
		AnimationTracks.JumpTrack:Stop()
		States.jumping = false
	end

	if Jumpgap > FatigueReset then --reset fatigue
		CurrJumpPower  = JumpPower
	end

	--Change to fall track if falling fast
	if verticleSpeed < -170 then
		States.jumping = false
		AnimationTracks.JumpTrack:Stop()
		Player:SetAttribute("SoundType", "Fall")
	end

	--print(HumState)

	utilities.ChangeFov(GetAllSpeed(Hrp)/2 + 70)


	if (Player:GetAttribute("Stamina") <= 0 and States.running) or States.rest then
		EndRun()
	end


	if HumState == Enum.HumanoidStateType.Jumping and not canDjump then
		Player:SetAttribute("SoundType", "DJump")
		print("jumpin")
	elseif HumState == Enum.HumanoidStateType.Jumping then
		Player:SetAttribute("SoundType", "Jump")	

	elseif States.running and Player:GetAttribute("Stamina") > 0 then

		SetWalkSpeed(2)
		if HumState ~= Enum.HumanoidStateType.Freefall and HumState ~= Enum.HumanoidStateType.Jumping and HorizontalSpeed > 1 then
			Player:SetAttribute("SoundType", "Run")
		elseif HumState ~= Enum.HumanoidStateType.Jumping then
			--Player:SetAttribute("SoundType", "None")
		end

	else

		SetWalkSpeed(1)
		if  HumState ~= Enum.HumanoidStateType.Freefall and HumState ~= Enum.HumanoidStateType.Jumping and HorizontalSpeed > 1 then
			Player:SetAttribute("SoundType", "Walk")
		elseif HumState ~= Enum.HumanoidStateType.Jumping then
			--Player:SetAttribute("SoundType", "None")
		end
	end

	if (HumState == Enum.HumanoidStateType.Freefall and HumState ~= Enum.HumanoidStateType.Jumping) or HorizontalSpeed < 1 then
		Player:SetAttribute("SoundType", "None")
	end

	prvState = HumState

end)
